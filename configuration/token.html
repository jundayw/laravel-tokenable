# 默认令牌驱动程序

/*
|--------------------------------------------------------------------------
| 默认令牌驱动程序
|---------------------------------------------------------------------------
|
| 此选项控制 TokenManager 使用的默认令牌驱动程序。
| 您可以选择“hash”或“jwt”，或者注册您自己的驱动程序。
|
*/

'default' => [
    'driver' => env('TOKEN_DRIVER', 'jwt'),
],

TIP

use Illuminate\Support\Facades\Auth;

Auth::guard('api')->login($user)->withToken('hash')->createToken('Hash Token');
